## 🗄️ 数据库初始化

### 步骤一：创建数据库

```sql
-- 1. 创建数据库
CREATE DATABASE qingshou CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 2. 创建专用用户
CREATE USER 'qingshou'@'localhost' IDENTIFIED BY 'QingShou@2026!';

-- 3. 授权
GRANT ALL PRIVILEGES ON qingshou.* TO 'qingshou'@'localhost';

-- 4. 刷新权限
FLUSH PRIVILEGES;
```

### 步骤二：执行建表SQL

将以下SQL依次执行（来自 07-MVP数据库设计与Java代码.md 中的DDL语句）：

```sql
-- 切换到轻瘦数据库
USE qingshou;

-- 用户表
CREATE TABLE `t_user` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键id',
  `username` varchar(50) NOT NULL DEFAULT '' COMMENT '用户名',
  `phone` varchar(20) NOT NULL DEFAULT '' COMMENT '手机号',
  `password` varchar(255) NOT NULL DEFAULT '' COMMENT '密码(加密存储)',
  `nickname` varchar(50) NOT NULL DEFAULT '' COMMENT '昵称',
  `avatar_url` varchar(500) NOT NULL DEFAULT '' COMMENT '头像URL',
  `gender` tinyint(1) NOT NULL DEFAULT '0' COMMENT '性别(0.未知,1.男,2.女)',
  `birthday` date DEFAULT NULL COMMENT '出生日期',
  `height` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '身高(cm)',
  `current_weight` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '当前体重(kg)',
  `target_weight` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '目标体重(kg)',
  `bmi` decimal(4,2) NOT NULL DEFAULT '0.00' COMMENT 'BMI指数',
  `activity_level` tinyint(1) NOT NULL DEFAULT '1' COMMENT '活动等级(1.久坐,2.轻度活动,3.中度活动,4.重度活动)',
  `calorie_goal` int(11) NOT NULL DEFAULT '1400' COMMENT '每日热量目标(kcal)',
  `user_status` tinyint(1) NOT NULL DEFAULT '1' COMMENT '用户状态(1.正常,2.冻结,3.注销)',
  `last_login_time` datetime DEFAULT NULL COMMENT '最后登录时间',
  `create_user` varchar(50) NOT NULL DEFAULT '' COMMENT '创建人Ad',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_user` varchar(50) NOT NULL DEFAULT '' COMMENT '修改人Ad',
  `edit_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  `delete_flag` tinyint(1) NOT NULL DEFAULT '0' COMMENT '删除标记(0.未删除,1.已删除)',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uniq_phone` (`phone`),
  KEY `idx_username` (`username`),
  KEY `idx_status` (`user_status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 体重记录表
CREATE TABLE `t_weight_record` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键id',
  `user_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '用户id',
  `record_date` date NOT NULL COMMENT '记录日期',
  `weight` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '体重(kg)',
  `body_fat_rate` decimal(4,2) NOT NULL DEFAULT '0.00' COMMENT '体脂率(%)',
  `muscle_rate` decimal(4,2) NOT NULL DEFAULT '0.00' COMMENT '肌肉率(%)',
  `water_rate` decimal(4,2) NOT NULL DEFAULT '0.00' COMMENT '水分率(%)',
  `waist_circumference` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '腰围(cm)',
  `hip_circumference` decimal(5,2) NOT NULL DEFAULT '0.00' COMMENT '臀围(cm)',
  `record_type` tinyint(1) NOT NULL DEFAULT '1' COMMENT '记录类型(1.手动录入,2.智能设备同步)',
  `remark` varchar(200) NOT NULL DEFAULT '' COMMENT '备注',
  `create_user` varchar(50) NOT NULL DEFAULT '' COMMENT '创建人Ad',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_user` varchar(50) NOT NULL DEFAULT '' COMMENT '修改人Ad',
  `edit_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  `delete_flag` tinyint(1) NOT NULL DEFAULT '0' COMMENT '删除标记(0.未删除,1.已删除)',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uniq_user_date` (`user_id`, `record_date`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_record_date` (`record_date`)

-- 餐食打卡表
CREATE TABLE `t_meal_record` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键id',
  `user_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '用户id',

  `meal_time` time NOT NULL COMMENT '用餐时间',
  `meal_content` varchar(500) NOT NULL DEFAULT '' COMMENT '餐食内容描述',
  `calorie_estimate` int(11) NOT NULL DEFAULT '0' COMMENT '热量估算(kcal)',
  `mood_type` tinyint(1) NOT NULL DEFAULT '0' COMMENT '心情类型(0.正常,1.开心,2.疲惫,3.压力大,4.其他)',
  `attachment_ids` varchar(500) NOT NULL DEFAULT '' COMMENT '附件id列表(逗号分隔)',
  `record_source` tinyint(1) NOT NULL DEFAULT '1' COMMENT '记录来源(1.手动录入,2.拍照识别)',
  `is_healthy` tinyint(1) NOT NULL DEFAULT '1' COMMENT '是否健康(0.不健康,1.健康)',
  `remark` varchar(200) NOT NULL DEFAULT '' COMMENT '备注',
  `create_user` varchar(50) NOT NULL DEFAULT '' COMMENT '创建人Ad',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_user` varchar(50) NOT NULL DEFAULT '' COMMENT '修改人Ad',
  `edit_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  `delete_flag` tinyint(1) NOT NULL DEFAULT '0' COMMENT '删除标记(0.未删除,1.已删除)',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_record_date` (`record_date`),
  KEY `idx_meal_type` (`meal_type`),
  KEY `idx_user_date_type` (`user_id`, `record_date`, `meal_type`)

-- 附件表
CREATE TABLE `t_attachment` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键id',
  `user_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '用户id',
  `file_name` varchar(255) NOT NULL DEFAULT '' COMMENT '文件名',
  `file_size` bigint(20) NOT NULL DEFAULT '0' COMMENT '文件大小(字节)',
  `file_type` varchar(50) NOT NULL DEFAULT '' COMMENT '文件类型',

  KEY `idx_attachment_type` (`attachment_type`),
  KEY `idx_create_time` (`create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='附件表';
```

## ⚙️ 后端项目启动

### 步骤一：克隆并配置

```bash
# 1. 进入项目目录
cd qingshou-backend

# 2. 复制配置文件
copy src/main/resources/application.yml.example src/main/resources/application.yml

# 3. 修改 application.yml 中的数据库密码、Redis密码等配置
notepad src/main/resources/application.yml
```

### 步骤二：Maven构建

```bash
# 清理并安装依赖
mvn clean install -DskipTests

# 或只安装不跳过测试
mvn clean install
```

### 步骤三：启动服务

```bash
# 方式一：Maven启动
mvn spring-boot:run

# 方式二：Java命令启动
java -jar target/qingshou-app-1.0.0-SNAPSHOT.jar

# 方式三：指定配置文件启动
java -jar target/qingshou-app-1.0.0-SNAPSHOT.jar --spring.profiles.active=dev
```

### 步骤四：验证启动

```bash
# 检查服务是否正常
curl http://localhost:8080/actuator/health

# 访问API文档
# 浏览器打开: http://localhost:8080/doc.html
```

## 📱 前端项目启动

### 步骤一：安装依赖

```bash
# 进入前端项目目录
cd qingshou-app

# 安装依赖
npm install

# 或使用 pnpm (推荐)
pnpm install
```

### 步骤二：配置环境变量

```bash
# 创建本地环境配置
copy .env.example .env.local

# 修改API地址
# 修改 .env.local 文件中的 VITE_API_BASE_URL=http://localhost:8080
notepad .env.local
```

`.env.local` 内容示例：
```
VITE_API_BASE_URL=http://localhost:8080
VITE_ENV=development
VITE_APP_TITLE=轻瘦(开发环境)
```

### 步骤三：启动开发服务

```bash
# H5版本
npm run dev

# 微信小程序版本
npm run dev:mp-weixin

# 安卓版本 (需要HBuilderX)
# 在HBuilderX中：运行 → 运行到手机或模拟器
```

### 步骤四：打包构建

```bash
# H5版本构建
npm run build

# 微信小程序构建
npm run build:mp-weixin
```

## 🐳 Docker快速部署（可选）

### docker-compose.yml

```yaml
version: '3.8'

services:
  # MySQL数据库
  mysql:
    image: mysql:8.0
    container_name: qingshou-mysql
    environment:
      MYSQL_ROOT_PASSWORD: root123
      MYSQL_DATABASE: qingshou
      MYSQL_USER: qingshou
      MYSQL_PASSWORD: QingShou@2026!
    volumes:
      - ./data/mysql:/var/lib/mysql
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    ports:
      - "3306:3306"
    networks:
      - qingshou-net
    restart: always

  # Redis缓存
  redis:
    image: redis:7.0
    container_name: qingshou-redis
    command: redis-server --requirepass redis123
    volumes:
      - ./data/redis:/data
    ports:
      - "6379:6379"
    networks:
      - qingshou-net
    restart: always

  # 后端应用
  app:
    build:
      context: ./qingshou-backend
      dockerfile: Dockerfile
    container_name: qingshou-app
    environment:
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/qingshou?useUnicode=true&characterEncoding=UTF-8&serverTimezone=Asia/Shanghai&useSSL=false
      SPRING_DATASOURCE_USERNAME: qingshou
      SPRING_DATASOURCE_PASSWORD: QingShou@2026!
      SPRING_REDIS_HOST: redis
      SPRING_REDIS_PASSWORD: redis123
    ports:
      - "8080:8080"
    depends_on:
      - mysql
      - redis
    networks:
      - qingshou-net
    restart: always

  # Nginx反向代理
  nginx:
    image: nginx:alpine
    container_name: qingshou-nginx
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/conf.d/default.conf
      - ./dist:/usr/share/nginx/html
    ports:
      - "80:80"
      - "443:443"
    depends_on:
      - app
    networks:
      - qingshou-net
    restart: always

networks:
  qingshou-net:
    driver: bridge
```

### Dockerfile (后端)

```dockerfile
# 构建阶段
FROM maven:3.9-openjdk-17 AS builder
WORKDIR /app
COPY pom.xml .
RUN mvn dependency:go-offline
COPY src ./src
RUN mvn clean package -DskipTests

# 运行阶段
FROM openjdk:17-jdk-slim
WORKDIR /app
COPY --from=builder /app/target/qingshou-app-*.jar app.jar
EXPOSE 8080
ENTRYPOINT ["java", "-Xms256m", "-Xmx512m", "-jar", "app.jar"]
```

### 启动命令

```bash
# 启动所有服务
docker-compose up -d

# 查看日志
docker-compose logs -f app

# 停止服务
docker-compose down
```
